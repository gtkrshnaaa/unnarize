ucoreUon

Built-in database with UON (Unnarize Object Notation) format.

Functions

FunctionDescription
load(file)Load UON database file
get(table)Get cursor for table
next(cursor)Get next row
insert(table, data)Insert data
save(file)Save to file

Loading Data

var success = ucoreUon.load("database.uon");

if (!success) {
    print("Failed to load database");
}

Querying Data

var cursor = ucoreUon.get("users");

var row = ucoreUon.next(cursor);
while (row) {
    print("User: " + row.name);
    row = ucoreUon.next(cursor);
}

UON File Format

// database.uon

[users]
{ name: "Alice", age: 30 }
{ name: "Bob", age: 25 }

[settings]
{ key: "theme", value: "dark" }
{ key: "version", value: "1.0" }

Complete Example

ucoreUon.load("app.uon");

print("=== Users ===");
var userCursor = ucoreUon.get("users");
var user = ucoreUon.next(userCursor);
while (user) {
    print(user.name + " - Age: " + user.age);
    user = ucoreUon.next(userCursor);
}

print("=== Settings ===");
var settingsCursor = ucoreUon.get("settings");
var setting = ucoreUon.next(settingsCursor);
while (setting) {
    print(setting.key + ": " + setting.value);
    setting = ucoreUon.next(settingsCursor);
}
