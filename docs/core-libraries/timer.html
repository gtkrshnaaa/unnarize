ucoreTimer

Timing operations and delays.

Functions

FunctionDescription
now()Current time in milliseconds
sleep(ms)Pause execution

Measuring Time

var start = ucoreTimer.now();

// Do some work
var sum = 0;
for (var i = 0; i < 1000000; i = i + 1) {
    sum = sum + i;
}

var elapsed = ucoreTimer.now() - start;
print("Elapsed: " + elapsed + "ms");

Delays

print("Starting...");
ucoreTimer.sleep(1000);  // Wait 1 second
print("1 second later");

ucoreTimer.sleep(500);   // Wait 500ms
print("Done!");

Benchmarking

function benchmark(name, iterations) {
    var start = ucoreTimer.now();
    
    for (var i = 0; i < iterations; i = i + 1) {
        // Work
    }
    
    var elapsed = ucoreTimer.now() - start;
    var opsPerSec = iterations / (elapsed / 1000);
    
    print(name + ": " + opsPerSec + " ops/sec");
}
